select species.common_names, states.id, species.scientific_name, categories.category, families.family, orders.order_name, parks.park_name, species.nativeness 
from species, categories, orders, families, parks, states
 where species.common_names like '%Gray%'
 and species.category_id = categories.id
 and categories.category like '%Mammal%'
 and species.order_id = orders.id
 and orders.order_name like '%Carnivora%'
 and species.family_id = families.id
 and families.family like '%Canidae%' 
 and parks.park_code = species.park_code
 and species.park_code like '%%'
 and parks.state_code like concat('%', states.id, '%');
